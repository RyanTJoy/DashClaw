  ---
  Connect to OpenClaw-Pro Dashboard

  Step 1: Copy the SDK file into your workspace. The file is at:
  sdk/openclaw-agent.js
  Place it somewhere like tools/openclaw-pro/openclaw-agent.js

  Step 2: Create a tracker module (e.g. tools/openclaw-pro/tracker.js):

  import { OpenClawAgent } from './openclaw-agent.js';

  const claw = new OpenClawAgent({
    baseUrl: 'https://openclaw-pro.vercel.app',
    apiKey: process.env.OPENCLAW_API_KEY,
    agentId: 'moltfire',
    agentName: 'MoltFire'
  });

  export default claw;

  Step 3: Add to your workspace .env:
  OPENCLAW_API_KEY=<Wes will provide this value>

  Step 4: Log significant actions:

  import claw from './tools/openclaw-pro/tracker.js';

  // Start a task
  const { action_id } = await claw.createAction({
    action_type: 'deploy',
    declared_goal: 'What you are doing',
    systems_touched: ['system1', 'system2'],
    risk_score: 20,
    confidence: 75
  });

  // Log an assumption
  await claw.registerAssumption({
    action_id,
    assumption: 'What you assumed',
    basis: 'Why you assumed it'
  });

  // When something needs follow-up
  await claw.registerOpenLoop({
    action_id,
    loop_type: 'dependency',
    description: 'What needs resolving',
    priority: 'medium'
  });

  // When done
  await claw.updateOutcome(action_id, {
    status: 'completed',
    output_summary: 'What happened'
  });

  // Or if it failed
  await claw.updateOutcome(action_id, {
    status: 'failed',
    error_message: 'What went wrong'
  });

  Action types: build, deploy, research, fix, config, test, message, api, review, refactor, monitor, alert, cleanup,
   sync, migrate, other

  What to log: Installs, deploys, config changes, decisions with assumptions, failures. Don't log: routine file
  reads, simple messages, every tool call.

  Verify connectivity:
  const result = await claw.createAction({
    action_type: 'test',
    declared_goal: 'Verify OpenClaw-Pro connectivity'
  });
  console.log('Connected! Action ID:', result.action_id);

  Dashboard: https://openclaw-pro.vercel.app
